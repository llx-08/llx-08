最速梯度下降

2020年03月07日 289 字 大概 1 分钟

最速传说（xxx

import numpy as np
import matplotlib.pyplot as plt
import math

def func(x):
    return 2*x - 1

def Random_Point(n):
    Points = np.random.rand(n) * 15 #噪声数据
    #print(Points)
    for i in range(n):
        Points[i] += func(i)

    return Points

def sgd(points,  start_w, start_b, alpha, epsi):
    x0 = np.full(20, 1.0) #因为是拟合直线，所以只需要表示成为y = w0*x0 + w1*x1
    x1 = np.linspace(0, 19, 20)

    data = np.vstack([x0, x1]).T #转换为对应格式
    parameter = [start_w, start_b] #起始参数
    #print(parameter[0], parameter[1])
    para_prev = np.zeros(2)

    sums = np.zeros(2)

    while np.linalg.norm(parameter - para_prev) > epsi:#两次更新参数之间的距离小于epsilon时停止迭代
        para_prev = parameter

        for i in range(20):
            sums += (np.dot(data[i], parameter) - points[i]) * data[i]#dot是矩阵点积，就是计算出结果减去真实值

        parameter -= alpha * sums/(2*20)#不除以2*m也无所谓，反正是常数，就当改变了学习率

    print("end with ", parameter)

    x = np.linspace(0, 19, 20)
    temp = []

    for i in range(20):
        plt.scatter(i, points[i])
        temp.append(np.dot(data[i], parameter))

    #print("temp:",temp)
    plt.plot(x, temp)
    plt.show()


start_w = 1
start_b = 3
print("parameter start with ",start_w," ",start_b)

sgd(Random_Point(20), start_w, start_b, 0.0001, 0.00000001)

仅供参考，参数可以改成自己输入
