Sort_with_Swap(0,i)

2020年03月02日 395 字 大概 2 分钟

oj

初始代码，结果超时了，思路和下面一份差不多，都写在注释里面了。

#include <bits/stdc++.h>
using namespace std;
vector<int> data;

int find(int a){
    int pos = 0;
    while (pos < data.size() and a != data[pos]) pos++;

    return pos;
}

int main(){
    int N;
    int ans = 0;

    cin>>N;

    for (int i = 0; i < N; i++){
        int temp;
        cin>>temp;
        data.push_back(temp);
    }

    while (1){
        int i = find(0);

        if (i == 0){
            int count = 0;
            while (count < N and data[count] == count)
                count++;
            if (count == N)
                break;
            swap(data[i], data[count]);

            ans++;
        } else{
            int k = find(i);
            swap(data[i], data[k]);
            ans++;
        }
    }
    cout<<ans;

    return 0;
}

借鉴了https://blog.csdn.net/qq_40941722/article/details/104505534?fps=1&locationNum=2的思路

用map只存没归位的数据，减去了查找的时间，在后面考虑0在0处的情况也不需要查找，下一个就是。

#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    vector<int> vec(100000);
    map<int,int> mp;
    cin>>n;
    for(int i=0; i<n; i++){
        int t;
        cin>>t;
        vec[i] = t;
        if(t != i)    // map只存储未归位的数据
            mp[t] = i;
    }
    int ans = 0;
    while(mp.size() > 1){    // map中只有0时结束
        if(vec[0] != 0){//0位不是0，则将应该在0所在的位置的元素与0交换
            int i = mp[0], j = mp[i];
            vec[i] = vec[j], vec[j] = 0;
            mp.erase(vec[i]);    // 数据归位，从字典中删除
            mp[0] = j;         // 修改0的位置
        }
        else{//0位是0，则寻找下一个不在其位置上的元素
            auto it = ++mp.begin(); // map.begin()为0，故需要加一，map只存了不在其位置上的，加1就行
            int i = it->second, t = it->first;
            vec[0] = t, vec[i] = 0, mp[t] = 0, mp[0] = i;
        }
        ans++;

//        for(int i=0; i<n; i++)
//            cout<<vec[i];
//        cout<<vec[n]<<endl;
    }
    cout<<ans<<endl;
    return 0;
}
