给定一棵二叉搜索树的先序遍历序列，要求你找出任意两结点的最近公共祖先结点（简称 LCA）。

输入格式：

输入的第一行给出两个正整数：待查询的结点对数 M（≤ 1 000）和二叉搜索树中结点个数 N（≤ 10 000）。随后一行给出 N 个不同的整数，为二叉搜索树的先序遍历序列。最后 M 行，每行给出一对整数键值 U 和 V。所有键值都在整型int范围内。

输出格式：

对每一对给定的 U 和 V，如果找到 A 是它们的最近公共祖先结点的键值，则在一行中输出 LCA of U and V is A.。但如果 U 和 V 中的一个结点是另一个结点的祖先，则在一行中输出 X is an ancestor of Y.，其中 X 是那个祖先结点的键值，Y 是另一个键值。如果 二叉搜索树中找不到以 U 或 V 为键值的结点，则输出 ERROR: U is not found. 或者 ERROR: V is not found.，或者 ERROR: U and V are not found.。

输入样例：

6 8
6 3 1 2 5 4 8 7
2 5
8 7
1 9
12 -3
0 8
99 99

输出样例：

LCA of 2 and 5 is 3.
8 is an ancestor of 7.
ERROR: 9 is not found.
ERROR: 12 and -3 are not found.
ERROR: 0 is not found.
ERROR: 99 and 99 are not found.

#include<bits/stdc++.h>
using namespace std;
const int maxn = 100000;

struct BSTree//二叉搜索树结构体
{
    int lt;//左
    int rt;//右
    int fa;//父
} BST[maxn];

int a[maxn],b[maxn];
int ans;//如果有公共节点，用来存放答案
int visitedRecord[maxn];//记录是否被访问过了

void build(int rt, int val)//build tree
{
    if(!rt)
        return;

    if(rt>val)//插左边
    {
        if(BST[rt].lt)//有左子树再判断
            build(BST[rt].lt, val);
        else
        {
            BST[rt].lt=val;
            BST[val].fa=rt;
            return;
        }
    }

    else//插右边
    {
        if(BST[rt].rt)
            build(BST[rt].rt, val);
        else
        {
            BST[rt].rt=val;
            BST[val].fa=rt;
            return ;
        }
    }
}

void F(int t)//找该节点的老父亲
{
    if(!t)
        return ;
    if(visitedRecord[t])
    {
        ans=t;//如果被标记了就说明被访问过了，即公共节点
        return ;
    }
    visitedRecord[t]=1;//第一次调用时把所有的经过的节点都标记
    F(BST[t].fa);//找到头
}

int main()
{
    int n,m;
    cin>>m>>n;

    for(int i=0; i<=10000; i++)//初始化
    {
        BST[i].lt=0;
        BST[i].rt=0;
        BST[i].fa=0;
    }

    for(int i=1; i<=n; i++)
    {
        cin>>a[i];
        b[i]=a[i];
    }

    sort(b+1,b+n+1);//为了后面的lower_bound
    int rt;//根节点

    for(int i=1; i<=n; i++)
    {
        a[i]=lower_bound(b+1, b+n+1, a[i])-b;//更新a[i]为b中a[i]的位置
        //cout<<"a:"<<a[i]<<" "<<a[i] + b<<endl;
        if(i==1)//第一个位置作为根节点
            rt=a[i];
        else
            build(rt, a[i]);
    }
    for(int i=1; i<=m; i++)
    {
        int st,ed;
        cin>>st>>ed;

        int s1=lower_bound(b+1,b+n+1,st)-b;
        int s2=lower_bound(b+1,b+n+1,ed)-b;

        if(b[s1]!=st and b[s2]!=ed)//两个都找不到{
            cout<<"ERROR: "<<st<<" and "<<ed<<" are not found."<<endl;

        else if(b[s1] != st)
            cout<<"ERROR: "<<st<<" is not found."<<endl;

        else if(b[s2] != ed)
            cout<<"ERROR: "<<ed<<" is not found."<<endl;

        else{
            F(s1);
            if(visitedRecord[s2])//s2在这条路径上
                cout<<ed<<" is an ancestor of "<<st<<"."<<endl;
            else
            {
                F(s2);
                if(ans==s1)
                    cout<<st<<" is an ancestor of "<<ed<<"."<<endl;
                else
                    cout<<"LCA of "<<st<<" and "<<ed<<" is "<<b[ans]<<"."<<endl;
            }
        }

        for(int i = 0; i < 10000;i++)//为下一次检测初始化
            visitedRecord[i] = 0;

        ans=0;
    }
}
